    <script>
        /* LÓGICA (El cerebro de la pantalla) */

        // 1. Detectar cambio de modalidad para ocultar/mostrar horarios
        function actualizarFormulario() {
            var modalidad = document.getElementById('modalidad').value;
            var seccionHorarios = document.getElementById('seccionHorarios');

            if (modalidad === 'asincronico') {
                seccionHorarios.style.display = 'none'; // Oculta los días si es asincrónico
            } else {
                seccionHorarios.style.display = 'block'; // Muestra los días si es presencial/virtual
            }
        }

        // 2. Simular el Guardado
        function guardarMateria() {
            var nombre = document.getElementById('nombreMateria').value;
            var modalidad = document.getElementById('modalidad').value;
            var asistencia = document.getElementById('asistencia').value;
            var mensajeDiv = document.getElementById('mensaje');

            if(nombre === "") {
                alert("Por favor, escribe el nombre de la materia.");
                return;
            }

            // Aquí es donde en el futuro conectaremos con la base de datos real.
            // Por ahora, le mostramos al usuario que el sistema "entendió".
            var textoConfirmacion = "<strong>¡Guardado!</strong><br>";
            textoConfirmacion += "Materia: " + nombre + "<br>";
            textoConfirmacion += "Modalidad: " + modalidad.toUpperCase() + "<br>";
            
            if (modalidad === 'asincronico') {
                textoConfirmacion += "Configuración: Se crearán recordatorios semanales.";
            } else {
                textoConfirmacion += "Configuración: Se controlará asistencia estricta del " + asistencia + "%.";
            }

            mensajeDiv.innerHTML = textoConfirmacion;
            mensajeDiv.style.display = 'block';
        }
    </script>
